# 事件追踪网站 - 部署方案与监控告警系统

## 1. 部署架构设计

### 1.1 整体部署架构

```
┌─────────────────────────────────────────────────────────────────┐
│                        用户访问层                                │
├─────────────────┬─────────────────┬───────────────────────────────┤
│    CloudFlare   │     AWS ELB     │      阿里云SLB               │
│   (全球CDN)     │   (负载均衡)    │    (负载均衡)                │
└─────────────────┴─────────────────┴───────────────────────────────┘
                              │
┌─────────────────────────────────────────────────────────────────┐
│                        应用服务层                                │
├─────────────────┬─────────────────┬───────────────────────────────┤
│   前端服务集群   │   API网关集群    │     微服务集群                │
│  (React SPA)   │   (Kong/Nginx)  │  (Docker + Kubernetes)        │
├─────────────────┼─────────────────┼───────────────────────────────┤
│  Web服务器 × 3   │  API网关 × 2    │ 认证服务 × 3 │ 事件服务 × 3  │
│  CDN加速       │  限流熔断       │ AI推送 × 2 │ 日历服务 × 2  │

                              │
┌─────────────────────────────────────────────────────────────────┐
│                        数据存储层                                │

├─────────────────┼─────────────────┼───────────────────────────────┤

│ 读写分离       │  哨兵模式       │  CDN加速  │ 跨区域备份       │
└─────────────────┴─────────────────┴───────────────────────────────┘
```

### 1.2 环境划分

#### 开发环境 (Development)
- **用途**: 日常开发、功能测试
- **规模**: 最小化部署，单实例
- **数据**: 模拟数据，可重置
- **访问**: 内部访问，VPN保护

#### 测试环境 (Testing/Staging)
- **用途**: 集成测试、回归测试
- **规模**: 中等规模，双实例
- **数据**: 测试数据，定期同步生产数据
- **访问**: 内部访问，权限控制

#### 预生产环境 (Pre-production)
- **用途**: 生产环境模拟、性能测试
- **规模**: 接近生产环境，全功能部署
- **数据**: 脱敏生产数据
- **访问**: 受限访问，审批流程

#### 生产环境 (Production)
- **用途**: 正式对外服务
- **规模**: 高可用集群，多区域部署
- **数据**: 真实用户数据，严格保护
- **访问**: 公网访问，安全防护

## 2. 容器化部署方案

### 2.1 Docker容器设计

#### 前端容器配置
```dockerfile
# apps/web/Dockerfile
FROM node:18-alpine AS builder

WORKDIR /app

# 安装依赖
COPY package*.json ./
RUN npm ci --only=production

# 构建应用
COPY . .
RUN npm run build

# 生产镜像
FROM nginx:alpine

# 复制构建产物
COPY --from=builder /app/dist /usr/share/nginx/html

# 复制nginx配置
COPY nginx.conf /etc/nginx/nginx.conf

# 健康检查
HEALTHCHECK --interval=30s --timeout=3s --start-period=5s --retries=3 \
  CMD curl -f http://localhost:80/health || exit 1

# 安全设置
RUN addgroup -g 1001 -S nginx && \
    adduser -S -D -H -u 1001 -h /var/cache/nginx -s /sbin/nologin -G nginx -g nginx nginx

USER nginx

EXPOSE 80

CMD ["nginx", "-g", "daemon off;"]
```

#### 后端容器配置
```dockerfile
# apps/api/Dockerfile
FROM node:18-alpine AS builder

WORKDIR /app

# 安装构建依赖
RUN apk add --no-cache python3 make g++

# 复制依赖文件
COPY package*.json ./
RUN npm ci

# 复制源码并构建
COPY . .
RUN npm run build

# 生产镜像
FROM node:18-alpine

WORKDIR /app

# 安装生产依赖
COPY package*.json ./
RUN npm ci --only=production && npm cache clean --force

# 复制构建产物
COPY --from=builder /app/dist ./dist
COPY --from=builder /app/prisma ./prisma

# 创建非root用户
RUN addgroup -g 1001 -S nodejs && \
    adduser -S -D -H -u 1001 -s /sbin/nologin -G nodejs -g nodejs nodejs

# 设置文件权限
RUN chown -R nodejs:nodejs /app
USER nodejs

# 健康检查
HEALTHCHECK --interval=30s --timeout=3s --start-period=5s --retries=3 \
  CMD node dist/healthcheck.js

EXPOSE 3000

# 启动应用
CMD ["node", "dist/index.js"]
```

#### 钉钉Webhook服务
```typescript
// infra/monitoring/dingtalk-webhook.js
const express = require('express');
const axios = require('axios');
const crypto = require('crypto');

const app = express();
app.use(express.json());

// 钉钉机器人配置
const dingtalkConfig = {
  critical: {
    webhook: process.env.DINGTALK_CRITICAL_WEBHOOK,
    secret: process.env.DINGTALK_CRITICAL_SECRET
  },
  warning: {
    webhook: process.env.DINGTALK_WARNING_WEBHOOK,
    secret: process.env.DINGTALK_WARNING_SECRET
  },
  info: {
    webhook: process.env.DINGTALK_INFO_WEBHOOK,
    secret: process.env.DINGTALK_INFO_SECRET
  }
};

// 生成签名
function generateSign(secret, timestamp) {
  const stringToSign = `${timestamp}\n${secret}`;
  const sign = crypto
    .createHmac('sha256', secret)
    .update(stringToSign)
    .digest('base64');
  return encodeURIComponent(sign);
}

// 发送钉钉消息
async function sendDingTalkMessage(webhook, secret, message) {
  const timestamp = Date.now();
  const sign = generateSign(secret, timestamp);
  
  const url = `${webhook}&timestamp=${timestamp}&sign=${sign}`;
  
  try {
    const response = await axios.post(url, message);
    console.log('钉钉消息发送成功:', response.data);
    return response.data;
  } catch (error) {
    console.error('钉钉消息发送失败:', error.message);
    throw error;
  }
}

// 格式化告警消息
function formatAlertMessage(alerts, severity) {
  const status = alerts[0].status;
  const title = status === 'firing' ? '🔥 告警触发' : '✅ 告警恢复';
  
  const colorMap = {
    critical: 'FF0000',
    warning: 'FFA500',
    info: '0080FF'
  };
  
  let text = `## ${title}\n\n`;
  text += `**告警级别:** ${severity.toUpperCase()}\n`;
  text += `**状态:** ${status === 'firing' ? '触发' : '恢复'}\n\n`;
  
  alerts.forEach(alert => {
    text += `---\n`;
    text += `**告警名称:** ${alert.labels.alertname}\n`;
    text += `**服务:** ${alert.labels.service || '未知'}\n`;
    text += `**实例:** ${alert.labels.instance || '未知'}\n`;
    text += `**描述:** ${alert.annotations.description || '无描述'}\n`;
    text += `**开始时间:** ${new Date(alert.startsAt).toLocaleString()}\n`;
    if (alert.endsAt) {
      text += `**结束时间:** ${new Date(alert.endsAt).toLocaleString()}\n`;
    }
  });
  
  return {
    msgtype: 'markdown',
    markdown: {
      title: `事件追踪系统${status === 'firing' ? '告警' : '恢复'}`,
      text: text
    },
    at: {
      isAtAll: severity === 'critical'
    }
  };
}

// Webhook端点
app.post('/dingtalk/:severity/send', async (req, res) => {
  try {
    const { severity } = req.params;
    const alerts = req.body.alerts || [];
    
    if (alerts.length === 0) {
      return res.status(400).json({ error: 'No alerts provided' });
    }
    
    const config = dingtalkConfig[severity];
    if (!config) {
      return res.status(400).json({ error: 'Invalid severity level' });
    }
    
    const message = formatAlertMessage(alerts, severity);
    await sendDingTalkMessage(config.webhook, config.secret, message);
    
    res.json({ success: true, message: 'Message sent successfully' });
  } catch (error) {
    console.error('Error sending DingTalk message:', error);
    res.status(500).json({ 
      success: false, 
      error: 'Failed to send message',
      details: error.message 
    });
  }
});

// 健康检查端点
app.get('/health', (req, res) => {
  res.json({ status: 'healthy', timestamp: new Date().toISOString() });
});

const PORT = process.env.PORT || 8060;
app.listen(PORT, () => {
  console.log(`钉钉Webhook服务运行在端口 ${PORT}`);
});
```

### 5.1 数据库备份方案

#### PostgreSQL自动备份脚本
```bash
#!/bin/bash
# infra/backup/postgres-backup.sh

set -e

# 配置变量
POSTGRES_HOST=${POSTGRES_HOST:-postgres}
POSTGRES_PORT=${POSTGRES_PORT:-5432}
POSTGRES_DB=${POSTGRES_DB:-event_tracker}
POSTGRES_USER=${POSTGRES_USER:-event_tracker}
BACKUP_DIR=${BACKUP_DIR:-/backups/postgres}
RETENTION_DAYS=${RETENTION_DAYS:-30}
S3_BUCKET=${S3_BUCKET:-event-tracker-backups}

# 创建备份目录
mkdir -p "$BACKUP_DIR"

# 生成备份文件名
BACKUP_FILE="${POSTGRES_DB}_$(date +%Y%m%d_%H%M%S).sql.gz"
BACKUP_PATH="$BACKUP_DIR/$BACKUP_FILE"

echo "开始备份PostgreSQL数据库: $POSTGRES_DB"

# 执行备份
docker exec event-tracker-postgres pg_dump \
  -h localhost \
  -p 5432 \
  -U "$POSTGRES_USER" \
  -d "$POSTGRES_DB" \
  --verbose \
  --clean \
  --if-exists \
  --create \
  --no-owner \
  --no-privileges \
  --no-tablespaces \
  --no-unlogged-table-data \
  --quote-all-identifiers \
  --compress=9 \
  --file=- | gzip > "$BACKUP_PATH"

# 检查备份是否成功
if [ $? -eq 0 ]; then
  echo "数据库备份成功: $BACKUP_PATH"
  
  # 上传到S3
  if command -v aws &> /dev/null; then
    echo "上传到S3..."
    aws s3 cp "$BACKUP_PATH" "s3://$S3_BUCKET/postgres/$BACKUP_FILE"
    
    if [ $? -eq 0 ]; then
      echo "备份文件已上传到S3: s3://$S3_BUCKET/postgres/$BACKUP_FILE"
    else
      echo "上传到S3失败"
      exit 1
    fi
  fi
  
  # 计算并记录备份文件大小
  BACKUP_SIZE=$(du -h "$BACKUP_PATH" | cut -f1)
  echo "备份文件大小: $BACKUP_SIZE"
  
  # 记录备份日志
  echo "$(date '+%Y-%m-%d %H:%M:%S') - 备份成功: $BACKUP_FILE (大小: $BACKUP_SIZE)" >> "$BACKUP_DIR/backup.log"
  
else
  echo "数据库备份失败"
  exit 1
fi

# 清理旧备份
echo "清理$RETENTION_DAYS天前的旧备份..."
find "$BACKUP_DIR" -name "*.sql.gz" -type f -mtime +$RETENTION_DAYS -delete

# 清理S3上的旧备份（如果配置了）
if command -v aws &> /dev/null; then
  echo "清理S3上的旧备份..."
  aws s3 ls "s3://$S3_BUCKET/postgres/" | while read -r line; do
    FILE_DATE=$(echo "$line" | awk '{print $1}')
    FILE_NAME=$(echo "$line" | awk '{print $4}')
    
    # 计算文件年龄（天数）
    CURRENT_DATE=$(date +%s)
    FILE_AGE=$(( (CURRENT_DATE - $(date -d "$FILE_DATE" +%s)) / 86400 ))
    
    if [ $FILE_AGE -gt $RETENTION_DAYS ]; then
      aws s3 rm "s3://$S3_BUCKET/postgres/$FILE_NAME"
      echo "已删除S3上的旧备份: $FILE_NAME"
    fi
  done
fi

echo "备份任务完成"
```

### 5.2 灾难恢复程序

#### 数据库恢复脚本
```bash
#!/bin/bash
# infra/disaster-recovery/postgres-restore.sh

set -e

# 配置变量
POSTGRES_HOST=${POSTGRES_HOST:-postgres}
POSTGRES_PORT=${POSTGRES_PORT:-5432}
POSTGRES_DB=${POSTGRES_DB:-event_tracker}
POSTGRES_USER=${POSTGRES_USER:-event_tracker}
BACKUP_FILE=${1:-"latest"}  # 备份文件路径或"latest"表示最新备份
S3_BUCKET=${S3_BUCKET:-event-tracker-backups}

# 颜色输出
RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
NC='\033[0m' # No Color

echo -e "${YELLOW}开始数据库恢复过程...${NC}"

# 获取备份文件
if [ "$BACKUP_FILE" = "latest" ]; then
  echo "获取最新备份文件..."
  
  # 从S3获取最新备份
  if command -v aws &> /dev/null; then
    LATEST_BACKUP=$(aws s3 ls "s3://$S3_BUCKET/postgres/" | sort | tail -n 1 | awk '{print $4}')
    
    if [ -z "$LATEST_BACKUP" ]; then
      echo -e "${RED}错误: 没有找到备份文件${NC}"
      exit 1
    fi
    
    echo "下载最新备份: $LATEST_BACKUP"
    aws s3 cp "s3://$S3_BUCKET/postgres/$LATEST_BACKUP" "/tmp/$LATEST_BACKUP"
    BACKUP_FILE="/tmp/$LATEST_BACKUP"
  else
    echo -e "${RED}错误: AWS CLI未安装，无法从S3获取备份${NC}"
    exit 1
  fi
fi

# 检查备份文件是否存在
if [ ! -f "$BACKUP_FILE" ]; then
  echo -e "${RED}错误: 备份文件不存在: $BACKUP_FILE${NC}"
  exit 1
fi

echo "使用备份文件: $BACKUP_FILE"

# 创建恢复确认提示
echo -e "${YELLOW}警告: 此操作将覆盖当前数据库的所有数据！${NC}"
echo -e "${YELLOW}数据库: $POSTGRES_DB${NC}"
echo -e "${YELLOW}主机: $POSTGRES_HOST:$POSTGRES_PORT${NC}"
echo -e "${YELLOW}备份文件: $BACKUP_FILE${NC}"
echo ""
read -p "确定要继续吗？(yes/no): " CONFIRM

if [ "$CONFIRM" != "yes" ]; then
  echo "恢复操作已取消"
  exit 0
fi

# 记录恢复开始时间
START_TIME=$(date +%s)

echo -e "${YELLOW}步骤1: 停止相关服务...${NC}"
# 停止应用服务（避免在恢复过程中写入数据）
docker-compose stop auth-service event-service ai-push-service

# 等待服务完全停止
sleep 10

echo -e "${YELLOW}步骤2: 创建数据库备份（安全措施）...${NC}"
SAFETY_BACKUP="/tmp/safety_backup_$(date +%Y%m%d_%H%M%S).sql.gz"
docker exec event-tracker-postgres pg_dump \
  -h localhost \
  -p 5432 \
  -U "$POSTGRES_USER" \
  -d "$POSTGRES_DB" \
  --verbose | gzip > "$SAFETY_BACKUP"

if [ $? -eq 0 ]; then
  echo -e "${GREEN}当前数据库备份创建成功: $SAFETY_BACKUP${NC}"
else
  echo -e "${RED}警告: 无法创建当前数据库备份，继续恢复操作...${NC}"
fi

echo -e "${YELLOW}步骤3: 删除现有数据库...${NC}"
docker exec event-tracker-postgres psql \
  -h localhost \
  -p 5432 \
  -U "$POSTGRES_USER" \
  -d postgres \
  -c "DROP DATABASE IF EXISTS $POSTGRES_DB;"

if [ $? -eq 0 ]; then
  echo -e "${GREEN}数据库删除成功${NC}"
else
  echo -e "${RED}错误: 无法删除数据库${NC}"
  exit 1
fi

echo -e "${YELLOW}步骤4: 创建新数据库...${NC}"
docker exec event-tracker-postgres psql \
  -h localhost \
  -p 5432 \
  -U "$POSTGRES_USER" \
  -d postgres \
  -c "CREATE DATABASE $POSTGRES_DB;"

if [ $? -eq 0 ]; then
  echo -e "${GREEN}数据库创建成功${NC}"
else
  echo -e "${RED}错误: 无法创建数据库${NC}"
  exit 1
fi

echo -e "${YELLOW}步骤5: 恢复数据库数据...${NC}"
# 解压备份文件（如果需要）
if [[ "$BACKUP_FILE" == *.gz ]]; then
  echo "解压备份文件..."
  gunzip -c "$BACKUP_FILE" | docker exec -i event-tracker-postgres psql \
    -h localhost \
    -p 5432 \
    -U "$POSTGRES_USER" \
    -d "$POSTGRES_DB"
else
  cat "$BACKUP_FILE" | docker exec -i event-tracker-postgres psql \
    -h localhost \
    -p 5432 \
    -U "$POSTGRES_USER" \
    -d "$POSTGRES_DB"
fi

if [ $? -eq 0 ]; then
  echo -e "${GREEN}数据库恢复成功${NC}"
else
  echo -e "${RED}错误: 数据库恢复失败${NC}"
  exit 1
fi

echo -e "${YELLOW}步骤6: 验证数据库完整性...${NC}"
# 检查数据库中的表
docker exec event-tracker-postgres psql \
  -h localhost \
  -p 5432 \
  -U "$POSTGRES_USER" \
  -d "$POSTGRES_DB" \
  -c "\dt"

if [ $? -eq 0 ]; then
  echo -e "${GREEN}数据库完整性检查通过${NC}"
else
  echo -e "${RED}警告: 数据库完整性检查失败${NC}"
fi

echo -e "${YELLOW}步骤7: 重启应用服务...${NC}"
docker-compose start auth-service event-service ai-push-service

# 等待服务启动
sleep 30

# 检查服务健康状态
echo "检查服务健康状态..."
docker-compose ps

END_TIME=$(date +%s)
DURATION=$((END_TIME - START_TIME))

echo -e "${GREEN}数据库恢复过程完成！${NC}"
echo -e "${GREEN}总耗时: $DURATION 秒${NC}"
echo -e "${GREEN}恢复的数据库: $POSTGRES_DB${NC}"
echo -e "${GREEN}使用的备份文件: $BACKUP_FILE${NC}"
echo -e "${GREEN}安全备份文件: $SAFETY_BACKUP${NC} (如需要可恢复)"

# 记录恢复日志
echo "$(date '+%Y-%m-%d %H:%M:%S') - 数据库恢复完成: $BACKUP_FILE (耗时: ${DURATION}s)" >> "/tmp/restore.log"
```

- **安全可靠性**: 保障系统安全，防止各类攻击
- **用户体验**: 确保良好的用户体验和易用性
- **兼容性**: 支持主流浏览器和设备

```typescript
    jest.clearAllMocks();

      { id: '1', title: '事件1' },

      total: 2
    });

    expect(result.current.loading).toBe(true);
    expect(result.current.events).toEqual([]);

    await waitForNextUpdate();

    expect(result.current.loading).toBe(false);
    expect(result.current.events).toEqual(mockEvents);
    expect(result.current.total).toBe(2);
  });

  it('应该正确处理事件创建', async () => {
    const newEvent = { title: '新事件', description: '描述' };
    const createdEvent = { id: '3', ...newEvent };
    
    (eventService.createEvent as jest.Mock).mockResolvedValue(createdEvent);

    const { result } = renderHook(() => useEvents());

    await act(async () => {
      const created = await result.current.createEvent(newEvent);
      expect(created).toEqual(createdEvent);
    });

    expect(eventService.createEvent).toHaveBeenCalledWith(newEvent);
  });

  it('应该正确处理错误状态', async () => {
    const error = new Error('获取失败');
    (eventService.getEvents as jest.Mock).mockRejectedValue(error);

    const { result, waitForNextUpdate } = renderHook(() => useEvents());
```

```typescript
// tests/performance/database-performance-test.ts
import { Client } from 'pg';
import { performance } from 'perf_hooks';

interface QueryPerformance {
  query: string;
  duration: number;
  rowCount: number;
  explain: any;
}

class DatabasePerformanceTest {
  private client: Client;
  private results: QueryPerformance[] = [];

  constructor(connectionString: string) {
    this.client = new Client({ connectionString });
  }

  async setup() {
    await this.client.connect();
    
    // 创建测试数据
    await this.createTestData();
  }

  async cleanup() {
    await this.client.end();
  }

  private async createTestData() {
    // 创建测试用户
    const userResult = await this.client.query(`
      INSERT INTO users (wechat_openid, nickname, email)
      VALUES ('test-openid', '测试用户', 'test@example.com')
      RETURNING id
    `);
    
    const userId = userResult.rows[0].id;
    
    // 创建测试事件（10万条）
    console.log('创建测试数据...');
    for (let i = 0; i < 100000; i++) {
      await this.client.query(`
        INSERT INTO events (user_id, title, description, type, priority, status, progress)
        VALUES ($1, $2, $3, $4, $5, $6, $7)
      `, [
        userId,
        `测试事件 ${i}`,
        `这是第 ${i} 个测试事件的描述`,
        i % 2 === 0 ? 'normal' : 'ai_push',
        ['high', 'medium', 'low'][i % 3],
        ['pending', 'in_progress', 'completed'][i % 3],
        Math.floor(Math.random() * 100)
      ]);
      
      if (i % 10000 === 0) {
        console.log(`已创建 ${i} 条事件`);
      }
    }
    
    console.log('测试数据创建完成');
  }

  async testQueryPerformance(query: string, params: any[] = []): Promise<QueryPerformance> {
    const startTime = performance.now();
    
    // 执行查询
    const result = await this.client.query(query, params);
    
    const endTime = performance.now();
    const duration = endTime - startTime;
    
    // 获取执行计划
    const explainResult = await this.client.query(`EXPLAIN (ANALYZE, BUFFERS) ${query}`, params);
    
    const performance: QueryPerformance = {
      query,
      duration,
      rowCount: result.rowCount,
      explain: explainResult.rows[0]
    };
    
    this.results.push(performance);
    
    return performance;
  }

  async runPerformanceTests() {
    console.log('开始数据库性能测试...');
    
    // 测试1: 简单查询
    await this.testQueryPerformance('
      SELECT COUNT(*) FROM events
    ');
    
    // 测试2: 带条件的查询
```

```typescript
// tests/security/dependency-scan.ts
import * as fs from 'fs';
import { execSync } from 'child_process';

interface Vulnerability {
  package: string;
  severity: 'low' | 'moderate' | 'high' | 'critical';
  title: string;
  url: string;
  vulnerableVersions: string;
  patchedVersions?: string;

class DependencySecurityScanner {
  private vulnerabilities: Vulnerability[] = [];

  async scanDependencies() {
    console.log('开始依赖安全扫描...');
    
    // 扫描前端依赖
    await this.scanFrontendDependencies();
    
    // 扫描后端依赖
    await this.scanBackendDependencies();
    
    // 扫描Docker镜像
    await this.scanDockerImages();
    
    console.log('依赖安全扫描完成');
  }

  private async scanFrontendDependencies() {
    console.log('扫描前端依赖...');
    
    try {
      const result = execSync('npm audit --json', { 
        cwd: './apps/web',
        encoding: 'utf8' 
      });
      
      const auditResult = JSON.parse(result);
      this.processNpmAuditResults(auditResult, 'frontend');
    } catch (error: any) {
      if (error.stdout) {
        const auditResult = JSON.parse(error.stdout);
        this.processNpmAuditResults(auditResult, 'frontend');
      }
    }
  }

  private async scanBackendDependencies() {
    console.log('扫描后端依赖...');
    
    try {
      const result = execSync('npm audit --json', { 
        cwd: './apps/api',
        encoding: 'utf8' 
      });
      
      const auditResult = JSON.parse(result);
      this.processNpmAuditResults(auditResult, 'backend');
    } catch (error: any) {
      if (error.stdout) {
        const auditResult = JSON.parse(error.stdout);
        this.processNpmAuditResults(auditResult, 'backend');
      }
    }
  }

  private async scanDockerImages() {
    console.log('扫描Docker镜像...');
    
    const images = [
      'event-tracker/web:latest',
      'event-tracker/api:latest',
      'event-tracker/auth-service:latest'
    ];
    
    for (const image of images) {
      try {
        // 使用Trivy进行容器镜像扫描
        const result = execSync(`trivy image --format json ${image}`, {
          encoding: 'utf8',
          maxBuffer: 10 * 1024 * 1024 // 10MB buffer
        });
        
        const trivyResult = JSON.parse(result);
        this.processTrivyResults(trivyResult, image);
      } catch (error) {
        console.error(`扫描镜像 ${image} 失败:`, error);
      }
    }
  }

  private processNpmAuditResults(result: any, component: string) {
    if (result.vulnerabilities) {
      Object.entries(result.vulnerabilities).forEach(([packageName, vuln]: [string, any]) => {
        this.vulnerabilities.push({
          package: `${component}:${packageName}`,
          severity: vuln.severity,
          title: vuln.title,
          url: vuln.url,
          vulnerableVersions: vuln.range,
          patchedVersions: vuln.fixAvailable ? vuln.fixAvailable.version : undefined
        });
      });
    }
  }

  private processTrivyResults(result: any, image: string) {
    if (result.Results) {
      result.Results.forEach((result: any) => {
        if (result.Vulnerabilities) {
          result.Vulnerabilities.forEach((vuln: any) => {
            this.vulnerabilities.push({
              package: `${image}:${vuln.PkgName}`,
              severity: vuln.Severity.toLowerCase(),
              title: vuln.Title,
              url: vuln.PrimaryURL,
              vulnerableVersions: vuln.InstalledVersion,
              patchedVersions: vuln.FixedVersion
            });
          });
        }
      });
    }
  }

  generateReport() {
    console.log('\n=== 依赖安全扫描报告 ===\n');
    
    const report = {
      summary: this.generateSummary(),
      vulnerabilities: this.vulnerabilities,
      recommendations: this.generateRecommendations()
    };
    
    // 保存报告
    fs.writeFileSync(
      'dependency-security-report.json',
      JSON.stringify(report, null, 2)
    );
    
    this.printSummary();
  }

  private generateSummary() {
    const total = this.vulnerabilities.length;
    const critical = this.vulnerabilities.filter(v => v.severity === 'critical').length;
    const high = this.vulnerabilities.filter(v => v.severity === 'high').length;
    const moderate = this.vulnerabilities.filter(v => v.severity === 'moderate').length;
    const low = this.vulnerabilities.filter(v => v.severity === 'low').length;

    return {
      total,
      bySeverity: { critical, high, moderate, low },
      riskLevel: this.calculateRiskLevel(critical, high),
      fixable: this.vulnerabilities.filter(v => v.patchedVersions).length
    };
  }

  private calculateRiskLevel(critical: number, high: number): string {
    if (critical > 0) return '极高风险';
    if (high > 5) return '高风险';
    if (high > 0) return '中等风险';
    return '低风险';
  }

  private generateRecommendations() {
    const recommendations = [];
    
    const criticalVulns = this.vulnerabilities.filter(v => v.severity === 'critical');
    if (criticalVulns.length > 0) {
      recommendations.push({
        priority: '紧急',
        issue: `发现 ${criticalVulns.length} 个严重漏洞`,
        action: '立即更新相关依赖包，部署安全补丁'
      });
    }
    
    const highVulns = this.vulnerabilities.filter(v => v.severity === 'high');
    if (highVulns.length > 0) {
      recommendations.push({
        priority: '高',
        issue: `发现 ${highVulns.length} 个高危漏洞`,
        action: '在下次发布前更新相关依赖包'
      });
    }
    
    const fixable = this.vulnerabilities.filter(v => v.patchedVersions).length;
    if (fixable > 0) {
      recommendations.push({
        priority: '中',
        issue: `${fixable} 个漏洞有可用的补丁`,
        action: '运行 npm audit fix 自动修复可修复的漏洞'
      });
    }
    
    return recommendations;
  }

  private printSummary() {

    console.log('\n=== 依赖安全扫描摘要 ===');

  }
```

```typescript
// tests/security/web-security-test.ts
import ZapClient from 'zaproxy';
import * as fs from 'fs';

interface SecurityIssue {
  alert: string;
  risk: 'High' | 'Medium' | 'Low' | 'Informational';
  confidence: 'High' | 'Medium' | 'Low' | 'False Positive';
  url: string;
  description: string;
  solution: string;
  reference: string;
}

class WebSecurityTester {
  private zap: ZapClient;
  private issues: SecurityIssue[] = [];
  private targetUrl: string;

  constructor(targetUrl: string) {
    this.targetUrl = targetUrl;
    this.zap = new ZapClient({
      proxy: 'http://localhost:8080'
    });
  }

  async runSecurityTests() {
    console.log(`开始对 ${this.targetUrl} 进行安全测试...`);
    
    try {
      // 启动ZAP扫描
      await this.startZAPScan();
      
      // 运行主动扫描
      await this.runActiveScan();
      
      // 运行被动扫描
      await this.runPassiveScan();
      
      // 获取扫描结果
      await this.getScanResults();
      
      // 生成报告
      this.generateReport();
      
    } catch (error) {
      console.error('安全测试失败:', error);
    }
  }

  private async startZAPScan() {
    console.log('启动ZAP扫描...');
    
    // 设置扫描策略
    await this.zap.ascan.setScannerPolicy('API-Scan');
    
    // 包含URL模式
    await this.zap.ascan.includeInContext('Default Context', `${this.targetUrl}.*`);
    
    // 排除静态资源
    await this.zap.ascan.excludeFromContext('Default Context', '.*\\.(js|css|png|jpg|gif|ico|svg)$');
    
    console.log('ZAP扫描配置完成');
  }

  private async runActiveScan() {
    console.log('运行主动安全扫描...');
    
    // 启动主动扫描
    const scanId = await this.zap.ascan.scan(this.targetUrl, true, false, null, null, null);
    
    console.log(`主动扫描ID: ${scanId}`);
    
    // 等待扫描完成
    let progress = 0;
    while (progress < 100) {
      progress = parseInt(await this.zap.ascan.status(scanId));
      console.log(`主动扫描进度: ${progress}%`);
      await this.sleep(5000); // 等待5秒
    }
    
    console.log('主动扫描完成');
  }

  private async runPassiveScan() {
    console.log('运行被动安全扫描...');
    
    // 访问目标网站以触发被动扫描
    await this.zap.core.accessUrl(this.targetUrl);
    
    // 等待被动扫描完成
    let recordsToScan = parseInt(await this.zap.pscan.recordsToScan());
    while (recordsToScan > 0) {
      console.log(`被动扫描剩余记录: ${recordsToScan}`);
      await this.sleep(2000);
      recordsToScan = parseInt(await this.zap.pscan.recordsToScan());
    }
    
    console.log('被动扫描完成');
  }

  private async getScanResults() {
    console.log('获取扫描结果...');
    
    // 获取所有告警
    const alerts = await this.zap.core.alerts(this.targetUrl);
    
    this.issues = alerts.map(alert => ({
      alert: alert.alert,
      risk: alert.risk,
      confidence: alert.confidence,
      url: alert.url,
      description: alert.description,
      solution: alert.solution,
      reference: alert.reference
    }));
    
    console.log(`发现 ${this.issues.length} 个安全问题`);

    );
    
    this.printSecuritySummary();

  private generateSecuritySummary() {
    const highRiskIssues = this.issues.filter(i => i.risk === 'High').length;
    const mediumRiskIssues = this.issues.filter(i => i.risk === 'Medium').length;
    const lowRiskIssues = this.issues.filter(i => i.risk === 'Low').length;
    const infoIssues = this.issues.filter(i => i.risk === 'Informational').length;

    return {
      totalIssues: this.issues.length,
      byRisk: {
        high: highRiskIssues,
        medium: mediumRiskIssues,
        low: lowRiskIssues,
        informational: infoIssues
      },
      riskLevel: this.calculateSecurityRiskLevel(highRiskIssues, mediumRiskIssues),
      topIssues: this.getTopSecurityIssues()
    };
  }

  private calculateSecurityRiskLevel(high: number, medium: number): string {
    if (high > 0) return '高风险';
    if (medium > 5) return '中等风险';
    if (medium > 0) return '低风险';
    return '安全';
  }

  private getTopSecurityIssues() {
    const riskScore = { 'High': 3, 'Medium': 2, 'Low': 1, 'Informational': 0 };
    
    return this.issues
      .sort((a, b) => riskScore[b.risk] - riskScore[a.risk])
      .slice(0, 10)
      .map(issue => ({
        alert: issue.alert,
        risk: issue.risk,
        url: issue.url,
        description: issue.description.substring(0, 100) + '...'
      }));
  }

  private generateSecurityRecommendations() {

    const highRiskIssues = this.issues.filter(i => i.risk === 'High');

    }

        priority: '高',
        issue: '发现SQL注入漏洞',
        action: '使用参数化查询，验证所有用户输入，实施最小权限原则'
      });
    }
    
    const xssIssues = this.issues.filter(i => 
      i.alert.toLowerCase().includes('cross site scripting')

    if (xssIssues.length > 0) {
      recommendations.push({
        priority: '高',
        issue: '发现XSS漏洞',
        action: '对所有输出进行HTML编码，实施内容安全策略(CSP)'
      });
    }
    
    return recommendations;
  }

  private printSecuritySummary() {

    console.log('\n=== 安全测试摘要 ===');
```

```typescript
        stats[result.browser] = { passed: 0, failed: 0 };

        stats[result.browser].passed++;
      } else {
        stats[result.browser].failed++;
      }
      
      return stats;
    }, {});

    return {
      totalTests: total,
      passed,
      failed,
      successRate: Math.round((passed / total) * 100),
      browserStats
```

```typescript
// tests/metrics/test-metrics.ts
import * as fs from 'fs';
import * as path from 'path';

interface TestMetrics {
  timestamp: string;
  testRun: {
    total: number;
    passed: number;
    failed: number;
    skipped: number;
    duration: number;
  };
  coverage: {
    lines: number;
    statements: number;
    functions: number;
    branches: number;
  };
  quality: {
    testPassRate: number;
    codeCoverage: number;
    defectDensity: number;
    testEffectiveness: number;
  };
  performance: {
    avgResponseTime: number;
    p95ResponseTime: number;
    errorRate: number;
    throughput: number;
  };
  security: {
    vulnerabilities: {
      critical: number;
      high: number;
      medium: number;
      low: number;
    };
    riskLevel: string;
  };
}

class TestMetricsCollector {
  private metrics: TestMetrics;

  constructor() {
    this.metrics = {
      timestamp: new Date().toISOString(),
      testRun: {
        total: 0,
        passed: 0,
        failed: 0,
        skipped: 0,
        duration: 0
      },
      coverage: {
        lines: 0,
        statements: 0,
        functions: 0,
        branches: 0
      },
      quality: {
        testPassRate: 0,
        codeCoverage: 0,
        defectDensity: 0,
        testEffectiveness: 0
      },
      performance: {
        avgResponseTime: 0,
        p95ResponseTime: 0,
        errorRate: 0,
        throughput: 0
      },
      security: {
        vulnerabilities: {
          critical: 0,
          high: 0,
          medium: 0,
          low: 0
        },
        riskLevel: 'unknown'
      }
    };
  }

  async collectMetrics() {
    console.log('开始收集测试度量指标...');
    
    // 收集单元测试指标
    await this.collectUnitTestMetrics();
    
    // 收集集成测试指标
    await this.collectIntegrationTestMetrics();
    
    // 收集性能测试指标
    await this.collectPerformanceMetrics();
    
    // 收集安全测试指标
    await this.collectSecurityMetrics();
    
    // 计算质量指标
    this.calculateQualityMetrics();
    
    // 生成报告
    this.generateMetricsReport();
    
    console.log('测试度量指标收集完成');
  }

  private async collectUnitTestMetrics() {
    // 读取Jest测试报告
    const jestReportPath = 'coverage/coverage-summary.json';
    if (fs.existsSync(jestReportPath)) {
      const jestReport = JSON.parse(fs.readFileSync(jestReportPath, 'utf8'));
      
      this.metrics.coverage = {
        lines: jestReport.total.lines.pct,
        statements: jestReport.total.statements.pct,
        functions: jestReport.total.functions.pct,
        branches: jestReport.total.branches.pct
      };
      
      // 读取测试执行结果
      const testResultsPath = 'test-results.json';
      if (fs.existsSync(testResultsPath)) {
        const testResults = JSON.parse(fs.readFileSync(testResultsPath, 'utf8'));
        this.metrics.testRun = {
          total: testResults.numTotalTests,
          passed: testResults.numPassedTests,
          failed: testResults.numFailedTests,
          skipped: testResults.numPendingTests,
          duration: testResults.testResults.reduce((sum: number, result: any) => 
            sum + result.endTime - result.startTime, 0
          )
        };
      }
    }
  }

  private async collectIntegrationTestMetrics() {
    // 读取集成测试报告
    const integrationReportPath = 'integration-test-results.json';
    if (fs.existsSync(integrationReportPath)) {
      const integrationReport = JSON.parse(fs.readFileSync(integrationReportPath, 'utf8'));
      
      // 合并到总体测试指标
      this.metrics.testRun.total += integrationReport.total;
      this.metrics.testRun.passed += integrationReport.passed;
      this.metrics.testRun.failed += integrationReport.failed;
      this.metrics.testRun.skipped += integrationReport.skipped;
    }
  }

  private async collectPerformanceMetrics() {
    // 读取K6性能测试报告
    const performanceReportPath = 'performance-report.json';
    if (fs.existsSync(performanceReportPath)) {
      const performanceReport = JSON.parse(fs.readFileSync(performanceReportPath, 'utf8'));
      
      this.metrics.performance = {
        avgResponseTime: performanceReport.metrics.http_req_duration.avg,
        p95ResponseTime: performanceReport.metrics.http_req_duration['p(95)'],
        errorRate: performanceReport.metrics.http_req_failed.rate,
        throughput: performanceReport.metrics.http_reqs.rate
      };
    }
  }

  private async collectSecurityMetrics() {
    // 读取安全扫描报告
    const securityReportPath = 'security-report.json';
    if (fs.existsSync(securityReportPath)) {
      const securityReport = JSON.parse(fs.readFileSync(securityReportPath, 'utf8'));
      
      this.metrics.security = {
        vulnerabilities: securityReport.summary.bySeverity,
        riskLevel: securityReport.summary.riskLevel
      };
    }
  }

  private calculateQualityMetrics() {
    const { total, passed, failed } = this.metrics.testRun;
    
    // 测试通过率
    this.metrics.quality.testPassRate = total > 0 ? (passed / total) * 100 : 0;
    
    // 代码覆盖率（取平均值）
    const coverage = this.metrics.coverage;
    this.metrics.quality.codeCoverage = (coverage.lines + coverage.statements + 
      coverage.functions + coverage.branches) / 4;
    
    // 缺陷密度（每千行代码的缺陷数）
    const totalLinesOfCode = this.estimateLinesOfCode();
    this.metrics.quality.defectDensity = totalLinesOfCode > 0 ? 
      (failed / totalLinesOfCode) * 1000 : 0;
    
    // 测试有效性（基于覆盖率和通过率）
    this.metrics.quality.testEffectiveness = 
      (this.metrics.quality.testPassRate * 0.6) + 
      (this.metrics.quality.codeCoverage * 0.4);
  }

  private estimateLinesOfCode(): number {
    // 这是一个简化的估算，实际项目中应该使用更准确的统计方法
    const sourceDirs = ['apps/web/src', 'apps/api/src'];
    let totalLines = 0;
    
    sourceDirs.forEach(dir => {
      if (fs.existsSync(dir)) {
        totalLines += this.countLinesInDirectory(dir);
      }
    });
    
    return totalLines;
  }

  private countLinesInDirectory(dir: string): number {
    let count = 0;
    const files = fs.readdirSync(dir, { withFileTypes: true });
    
    files.forEach(file => {
      const filePath = path.join(dir, file.name);
      
      if (file.isDirectory()) {
        count += this.countLinesInDirectory(filePath);
      } else if (file.name.endsWith('.ts') || file.name.endsWith('.tsx') || 
                 file.name.endsWith('.js') || file.name.endsWith('.jsx')) {
        const content = fs.readFileSync(filePath, 'utf8');
        count += content.split('\n').length;
      }
    });
    
    return count;
  }

  private generateMetricsReport() {
    console.log('\n=== 测试度量报告 ===\n');
    
    const report = {
      metrics: this.metrics,
      analysis: this.analyzeMetrics(),
      trends: this.generateTrends(),
      recommendations: this.generateMetricRecommendations()
    };
    
    // 保存报告
    fs.writeFileSync(
      'test-metrics-report.json',
      JSON.stringify(report, null, 2)
    );
    
    this.printMetricsSummary();
  }

  private analyzeMetrics() {
    const analysis = {
      testQuality: this.evaluateTestQuality(),
      performanceHealth: this.evaluatePerformanceHealth(),
      securityPosture: this.evaluateSecurityPosture(),
      overallHealth: 'unknown'
    };
    
    // 综合评估
    const scores = [
      analysis.testQuality.score,
      analysis.performanceHealth.score,
      analysis.securityPosture.score
    ];
    
    const overallScore = scores.reduce((sum, score) => sum + score, 0) / scores.length;
    
    if (overallScore >= 90) {
      analysis.overallHealth = '优秀';
    } else if (overallScore >= 80) {
      analysis.overallHealth = '良好';
    } else if (overallScore >= 70) {
      analysis.overallHealth = '一般';
    } else {
      analysis.overallHealth = '需要改进';
    }
    
    return analysis;
  }

  private evaluateTestQuality() {
    const passRate = this.metrics.quality.testPassRate;
    const coverage = this.metrics.quality.codeCoverage;
    
    let score = 0;
    let status = 'unknown';
    
    if (passRate >= 95 && coverage >= 90) {
      score = 100;
      status = '优秀';
    } else if (passRate >= 90 && coverage >= 80) {
      score = 85;
      status = '良好';
    } else if (passRate >= 80 && coverage >= 70) {
      score = 70;
      status = '一般';
    } else {
      score = 50;
      status = '需要改进';
    }
    
    return {
      score,
      status,
      details: {
        passRate,
        coverage,
        defectDensity: this.metrics.quality.defectDensity
      }
    };
  }

  private evaluatePerformanceHealth() {
    const avgResponseTime = this.metrics.performance.avgResponseTime;
    const errorRate = this.metrics.performance.errorRate;
    
    let score = 0;
    let status = 'unknown';
    
    if (avgResponseTime < 200 && errorRate < 1) {
      score = 100;
      status = '优秀';
    } else if (avgResponseTime < 500 && errorRate < 5) {
      score = 85;
      status = '良好';
    } else if (avgResponseTime < 1000 && errorRate < 10) {
      score = 70;
      status = '一般';
    } else {
      score = 50;
      status = '需要改进';
    }
    
    return {
      score,
      status,
      details: {
        avgResponseTime,
        p95ResponseTime: this.metrics.performance.p95ResponseTime,
        errorRate,
        throughput: this.metrics.performance.throughput
      }
    };
  }

  private evaluateSecurityPosture() {
    const vulnerabilities = this.metrics.security.vulnerabilities;
    const totalVulns = Object.values(vulnerabilities).reduce((sum, count) => sum + count, 0);
    
    let score = 0;
    let status = 'unknown';
    
    if (vulnerabilities.critical === 0 && vulnerabilities.high === 0) {
      score = 100;
      status = '优秀';
    } else if (vulnerabilities.critical === 0 && vulnerabilities.high <= 5) {
      score = 85;
      status = '良好';
    } else if (vulnerabilities.critical <= 2) {
      score = 70;
      status = '一般';
    } else {
      score = 30;
      status = '高风险';
    }
    
    return {
      score,
      status,
      details: {
        totalVulnerabilities: totalVulns,
        bySeverity: vulnerabilities,
        riskLevel: this.metrics.security.riskLevel
      }
    };
  }

  private generateTrends() {
    // 这里可以添加历史数据比较，生成趋势分析
    return {
      testPassRate: 'stable', // improving, stable, declining
      codeCoverage: 'improving',
      performance: 'stable',
      security: 'improving'
    };
  }

  private generateMetricRecommendations() {
    const recommendations = [];
    
    // 基于测试质量
    if (this.metrics.quality.testPassRate < 90) {
      recommendations.push({
        area: '测试质量',
        issue: '测试通过率较低',
        suggestion: '修复失败的测试用例，提高代码质量'
      });
    }
    
    if (this.metrics.quality.codeCoverage < 80) {
      recommendations.push({
        area: '测试质量',
        issue: '代码覆盖率不足',
        suggestion: '增加单元测试，覆盖核心业务逻辑'
      });
    }
    
    // 基于性能
    if (this.metrics.performance.avgResponseTime > 500) {
      recommendations.push({
        area: '性能',
        issue: '平均响应时间较长',
        suggestion: '优化数据库查询，添加缓存，优化API响应'
      });
    }
    
    // 基于安全
    const securityIssues = this.metrics.security.vulnerabilities;
    if (securityIssues.critical > 0 || securityIssues.high > 0) {
      recommendations.push({
        area: '安全',
        issue: '发现高危安全漏洞',
        suggestion: '立即修复安全漏洞，更新依赖包，加强安全监控'
      });
    }
    
    return recommendations;
  }

  private printMetricsSummary() {
    console.log('\n=== 测试度量摘要 ===');
    console.log(`测试通过率: ${this.metrics.quality.testPassRate.toFixed(1)}%`);
    console.log(`代码覆盖率: ${this.metrics.quality.codeCoverage.toFixed(1)}%`);
    console.log(`平均响应时间: ${this.metrics.performance.avgResponseTime.toFixed(0)}ms`);
    console.log(`P95响应时间: ${this.metrics.performance.p95ResponseTime.toFixed(0)}ms`);
    console.log(`错误率: ${this.metrics.performance.errorRate.toFixed(2)}%`);
    
    const securityVulns = this.metrics.security.vulnerabilities;
    console.log(`安全漏洞: ${Object.values(securityVulns).reduce((a, b) => a + b, 0)}个`);
    console.log(`  - 严重: ${securityVulns.critical}`);
    console.log(`  - 高危: ${securityVulns.high}`);
    console.log(`  - 中等: ${securityVulns.medium}`);
    console.log(`  - 低危: ${securityVulns.low}`);
    
    const analysis = this.analyzeMetrics();
    console.log(`\n整体健康状况: ${analysis.overallHealth}`);
    
    if (analysis.overallHealth === '需要改进') {
      console.log('\n⚠️  发现质量问题，需要立即改进！');
    }
  }
}
```
